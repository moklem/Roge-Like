```gdscript
extends CharacterBody2D
## Player movement controller — handles WASD input and wall collision.
## P3: All input handling guarded by is_multiplayer_authority().
## P4: MultiplayerSynchronizer replicates position at 20 Hz (replication_interval = 0.05).

const SPEED: float = 200.0

@export var peer_id: int = 0
@export var role_label: String = ""

func _ready() -> void:
    # Set authority based on peer_id — only the owning peer controls this player
    set_multiplayer_authority(peer_id)

func _physics_process(delta: float) -> void:
    # P3: Only the authority peer reads input and moves
    if not is_multiplayer_authority():
        return
    
    var dir := Input.get_vector("ui_left", "ui_right", "ui_up", "ui_down")
    velocity = dir * SPEED
    move_and_slide()
```

- Use `Input.get_vector("ui_left", "ui_right", "ui_up", "ui_down")` — these input actions were configured in Phase 1 project.godot

Update `scenes/Game.gd`:

```gdscript
extends Node2D
## Game scene controller — spawns players, manages room lifecycle.
## P7: All spawnable scene types registered in MultiplayerSpawner.

const PLAYER_SCENE := preload("res://scenes/Player.tscn")

func _ready() -> void:
    # Spawn local player (this peer) at spawn point
    _spawn_players()
    
    # Remove the back button — no way out once game starts (host disconnect ends game)
    
func _spawn_players() -> void:
    # Find spawn point for this peer
    var spawn_points := $Room1/SpawnPoints.get_children()
    var spawn_index := 0
    var my_id := multiplayer.get_unique_id()
    
    # Find our index in the players dict to pick spawn position
    var idx := 0
    for id in Lobby.players:
        if id == my_id:
            spawn_index = idx
            break
        idx += 1
    
    if spawn_index < spawn_points.size():
        var spawn_pos := spawn_points[spawn_index].global_position
        _spawn_player_at(my_id, spawn_pos)

func _spawn_player_at(id: int, pos: Vector2) -> void:
    var player := PLAYER_SCENE.instantiate()
    player.position = pos
    player.peer_id = id
    player.role_label = Lobby.players.get(id, {}).get("role", "Player")
    player.name = "Player_%s" % id  # force_readable_name for RPC (P1)
    $Room1/Entities.add_child(player, true)  # force_readable_name = true
```

| Client input → Host authority | Player input is read only on the authority peer; position is synced via MultiplayerSynchronizer (read-only on clients) |
| MultiplayerSpawner → Entities | Only host spawns nodes via spawner; clients receive auto-replicated add_child |
