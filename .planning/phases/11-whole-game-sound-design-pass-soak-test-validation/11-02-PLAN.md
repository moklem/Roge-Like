## Reserved-voice playback for must-hear stingers. Overflow policy = D-04: if every reserved
## voice is busy, STEAL a voice from the shared/routine pool (never drop, never cut off another
## priority stinger). Only if the shared pool is somehow empty do we fall back to a reserved slot.
func _play_priority(stream: AudioStream, volume_db: float, pitch_min: float, pitch_max: float) -> void:
    if stream == null:
        return
    var p: AudioStreamPlayer = null
    # 1. Prefer a free reserved voice.
    for cand in _priority_players:
        if not cand.playing:
            p = cand
            break
    # 2. D-04 overflow — all reserved voices busy: steal from the SHARED pool (_players), not from
    #    another reserved slot. Prefer a free shared voice; if none is free, forcibly reuse the
    #    routine round-robin slot _players[_next] and advance _next (cuts off at most one routine hit).
    if p == null and not _players.is_empty():
        for cand in _players:
            if not cand.playing:
                p = cand
                break
        if p == null:
            p = _players[_next]
            _next = (_next + 1) % _players.size()
    # 3. Last resort (no shared pool configured at all): round-robin a reserved slot.
    if p == null and not _priority_players.is_empty():
        p = _priority_players[_priority_next]
        _priority_next = (_priority_next + 1) % _priority_players.size()
    if p == null:
        return
    p.stream = stream
    p.volume_db = volume_db
    p.pitch_scale = randf_range(pitch_min, pitch_max)
    p.play()
```

## D-06: echo the CarHUD's car-framing into sound, but SFX-02 requires true onset-only.
## The Earth passive heal re-emits "seat_massage" every 1.0s while any Earth player is alive
## (Game.gd _tick_earth_effects), and Fire/Ice re-emit "engine"/"ac" in rapid bursts. A fixed
## rolling debounce would only THROTTLE these to a cue every ~1.5s — still machine-gunning.
## Instead detect a real ONSET: fire only when this event name has been quiet longer than
## HUD_ECHO_ONSET_GAP (the effect actually stopped and restarted). _hud_echo_last is advanced
## on EVERY event — including suppressed ones — so a continuous 1.0s stream keeps resetting the
## gap and never re-triggers until the effect stops for real. (The per-tick burn damage / heal
## itself emits no hud_event; only these onset echoes are ever heard.)
func _on_hud_event(event_name: String) -> void:
    var now: float = Time.get_unix_time_from_system()
    var last: float = float(_hud_echo_last.get(event_name, -1000.0))
    _hud_echo_last[event_name] = now
    if now - last < HUD_ECHO_ONSET_GAP:
        return
    match event_name:
        "ac": ice_ac_hiss()
        "engine": fire_engine_hiss()
        "seat_massage": earth_servo_hum()
        "lidar": lidar_blip()
```
