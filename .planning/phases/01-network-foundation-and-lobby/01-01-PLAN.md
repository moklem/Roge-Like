Create `autoloads/Lobby.gd`:
```gdscript
extends Node

const PORT: int = 7000        # D-10: hardcoded, no UI
const MAX_CLIENTS: int = 2    # max 2 clients; host is peer 1

# peer_id → {name: String, role: String, element: String, ready: bool}
var players: Dictionary = {}

func _ready() -> void:
    # D-16 / P9: wire peer_disconnected BEFORE any scene loads
    multiplayer.peer_connected.connect(_on_peer_connected)
    multiplayer.peer_disconnected.connect(_on_peer_disconnected)
    multiplayer.server_disconnected.connect(_on_server_disconnected)

func create_game() -> void:
    var peer := ENetMultiplayerPeer.new()
    var err := peer.create_server(PORT, MAX_CLIENTS)
    if err != OK:
        push_error("Lobby: failed to create server: %s" % error_string(err))
        return
    multiplayer.multiplayer_peer = peer
    # Register host in players dict immediately (D-13: host is always peer 1)
    players[1] = {name = "Tank", role = "", element = "", ready = false}

func join_game(ip: String) -> void:
    var peer := ENetMultiplayerPeer.new()
    var err := peer.create_client(ip, PORT)
    if err != OK:
        push_error("Lobby: failed to join: %s" % error_string(err))
        connection_failed.emit()
        return
    multiplayer.multiplayer_peer = peer

func remove_multiplayer_peer() -> void:
    multiplayer.multiplayer_peer = OfflineMultiplayerPeer.new()
    players.clear()

func _on_peer_connected(id: int) -> void:
    # P2: send current state to newly connected peer (not in _ready)
    _sync_player_list_to.rpc_id(id, players)

func _on_peer_disconnected(id: int) -> void:
    players.erase(id)
    # D-16 / P9: if host (id==1) disconnected, all clients return to main menu
    if id == 1:
        get_tree().change_scene_to_file("res://scenes/ui/GameOver.tscn")
    else:
        player_list_changed.emit()

func _on_server_disconnected() -> void:
    # Called on client when host disappears (server_disconnected fires before peer_disconnected)
    get_tree().change_scene_to_file("res://scenes/ui/GameOver.tscn")

# D-14/P1: All @rpc functions have IDENTICAL annotation + signature on both host and client.
# This function is defined once here; both host and client have this autoload loaded.

@rpc("any_peer", "call_remote", "reliable")
func _sync_player_list_to(_list: Dictionary) -> void:
    # Called from host to a specific newly connected peer
    players = _list
    player_list_changed.emit()

@rpc("authority", "call_local", "reliable")
func notify_player_updated(id: int, data: Dictionary) -> void:
    players[id] = data
    player_list_changed.emit()

@rpc("authority", "call_local", "reliable")
func notify_game_starting() -> void:
    game_starting.emit()

signal player_list_changed()
signal connection_failed()
signal game_starting()
```

    pass  # Minimal until Phase 6 wires loop timer and difficulty scaling

    if not multiplayer.is_server():
        return  # D-13: only host ticks timer
    if loop_timer > 0.0:
        loop_timer -= delta

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client → host (RPC) | Client sends role/element pick and ready state via RPC; host is sole validator |
| IP input → join | User enters arbitrary IP string; passed to ENet create_client |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-01 | Spoofing | Lobby.join_game IP input | accept | LAN-only demo; no authentication needed; invalid IPs fail at ENet level with error code |
| T-01-02 | Denial of Service | ENetMultiplayerPeer.create_server (port 7000) | accept | LAN demo context; no public exposure; OS firewall handles external access |
| T-01-03 | Elevation of Privilege | RPC `@rpc("any_peer")` functions | mitigate | Only `_sync_player_list_to` uses any_peer; restricted to one-way data sync from host; all state-changing RPCs use `@rpc("authority")` which limits callers to host only |
| T-01-04 | Information Disclosure | players Dictionary broadcast | accept | Contains only role/element/name — no PII, no secrets; shared with all lobby participants intentionally |
</threat_model>
